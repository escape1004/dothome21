04. switch문 

{
let num = 100;
                
switch (num) {

        document.write("*** 04. switch문 ***");
        document.write("조건문이 실행되었습니다.(num == 100)");
    break;
    case 200:
        document.write("*** 04. switch문 ***");
        document.write("조건문이 실행되었습니다.(num == 200)");
    break;
    case 300:
        document.write("*** 04. switch문 ***");
        document.write("조건문이 실행되었습니다.(num == 300)");
    break;
    default:
        document.write("*** 04. switch문 ***");
        document.write("조건문이 실행되었습니다.(num == XX)");
    break;
}

07. while문 

{
    document.write("*** 07. while문 ***");
    
    let num = 1;
    while (num <= 10) {
        document.write(num + ". 반복문이 실행되었습니다.");

    }
}

100. 구구단 만들기 

{
    //1~9 for(var i = 1; 1 <= 9; i++)

    //num1 * num2 = sum

    // 2 * 1 = 2;      3 * 1 = 3;
    // 2 * 2 = 4;      3 * 2 = 6; 
    // 2 * 3 = 6;      3 * 3 = 9;
    // 2 * 4 = 8;      ..........
    // 2 * 5 = 10;     9 * 1 = 9;
    // 2 * 6 = 12;     9 * 2 = 18;
    // 2 * 7 = 14;
    // 2 * 8 = 16;
    // 2 * 9 = 18;

    document.write("*** 100. 구구단 만들기 ***");

    for (var i = 2; i <= 9; i++) {
        for (var j = 1; j <= 9; j++) {
            var sum = i * j;
            document.write(i + " * " + j + " = " + sum);
        }
    }
}

101. 테이블 만들기 

{
    document.write("*** 101. 테이블 만들기 ***");

    let num = 1;
    let table = "<table>";
    
    for (let i = 1; i <= 10; i++) {
        table += "<tr>";
        for (let j = 1; j <= 10; j++) {
            table += "<td>" + num + "</td>";
            num++;
        }
        table += "</tr>";
    }
    
    table += "</table>";
    
    document.write(table);
}
