01. 선언적 함수

{
    function func() {
        //실행문
        document.write("*** 01. 선언적 함수 ***");
        document.write("함수가 실행되었습니다.");
    }
    func();
}

02. 익명 함수

{
    const func = function(){
        document.write("*** 02. 익명 함수 ***");
        document.write("함수가 실행되었습니다.");
    }
    func();
}

03. 매개변수 함수

{
    function func(str) {
        document.write("*** 03. 매개변수 함수 ***");
        document.write(str);
    }
    func("함수가 실행되었습니다.");
}

04. 리턴값 함수

{
    function func(){
        const str = "함수가 실행되었습니다.";
        return str;
    }
    document.write("*** 04. 리턴값 함수 ***");
    document.write(func());
}
