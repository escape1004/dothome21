    }

    function funB() {
        console.log("funB가 실행되었습니다.");
    }

    funB();
    funA();

    //02
    function funC() {
        // setTimeout(function () {
        //     document.write("funC가 실행되었습니다.");
        // }, 1000);

        setTimeout(() => {
            console.log("funC가 실행되었습니다.");
        }, 1000);
    }

    function funD() {
        console.log("funD가 실행되었습니다.");
    }

    funC();
    funD();

    //03
    function funE(callback) {
        setTimeout(() => {
            console.log("funE가 실행되었습니다.");
            callback();
        }, 1000);
    }

    function funF() {
        console.log("funF가 실행되었습니다.");
    }

    funE(funF);
}

    info1.result();
    info2.result();
}

1. 내 이름은 박세진이며, 직업은 웹 퍼블리셔입니다.
2. 내 이름은 세진이며, 직업은 프론트엔드 개발자입니다.
