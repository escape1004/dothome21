계층 선택자
번호 종류 jQuery Javascript 설명
2 하위 선택자 $("div p") querySelectorAll("li") div 태그 모든 자식중 p 태그를 모두 선택합니다.
자식 선택자 $("div > p") - div 태그 모든 자식 중 첫 번째 자식 p 태그만 선택합니다.
형제 선택자 $("div + p") - div 태그의 인접한 형태 p 태그 중 첫 번째 태그만 선택합니다.
형제들 선택자 $("div ~ p") - div 태그의 인접한 형제 p 태그 중 모든 형제 태그를 선택합니다.

01. 기본 선택자

  • $("li")
  • $(".class")
  • $("#id")
  • $(".class, #id")
  • $("*")
리셋 클릭하면 클래스(.class)를 선택합니다. 클릭하면 아이디(#id)를 선택합니다. 클릭하면 기본 태그(li)를 선택합니다. 클릭하면 여러가지 태그(#id, .class)를 선택합니다. 클릭하면 전체 태그(*)를 선택합니다. 
{
    $(".reset1").click(function () {
        $(".view1 *").removeClass("red");
    });
    $(".btn1-1").click(function () {
        $(".view1 .class").addClass("red");
    });
    $(".btn1-2").click(function () {
        $(".view1 #id").addClass("red");
    });
    $(".btn1-3").click(function () {
        $(".view1 li").addClass("red");
    });
    $(".btn1-4").click(function () {
        $(".view1 .class, .view #id").addClass("red");
    });
    $(".btn1-5").click(function () {
        $(".view1 *").addClass("red");
    });
}

01. 기본 선택자 - javascript

  • $("li")
  • $(".class")
  • $("#id")
  • $(".class, #id")
  • $("*")
리셋 클릭하면 클래스(.class)를 선택합니다. 클릭하면 아이디(#id)를 선택합니다. 클릭하면 기본 태그(li)를 선택합니다. 
{
    const jreset1 = document.querySelector(".jreset1");
    const jbtn1_1 = document.querySelector(".jbtn1-1");
    const jbtn1_2 = document.querySelector(".jbtn1-2");
    const jbtn1_3 = document.querySelector(".jbtn1-3");
    const jbtn1_4 = document.querySelector(".jbtn1-4");
    const jbtn1_5 = document.querySelector(".jbtn1-5");

    jreset1.addEventListener("click", function () {
        const liList = document.querySelectorAll(".jview1 *");
        for (let i = 0; i < liList.length; i++) {
            liList[i].classList.remove("red");
        }
    });

    jbtn1_1.addEventListener("click", function () {
        document.querySelector(".jview1 .j1class").classList.add("red");
    });

    jbtn1_2.addEventListener("click", function () {
        document.getElementById("j1id").classList.add("red");
    });

    jbtn1_3.addEventListener("click", function () {
        const liList = document.querySelectorAll(".jview1 li");
        //for문
        // for (i = 0; i < liList.length; i++) {
        //     liList[i].classList.add("red");
        // }

        //forEach문
        // liList.forEach(el => {
        //     el.classList.add("red");
        // });

        //for of문
        // for (let i of liList) {
        //     i.classList.add("red");
        // }

        //for in문
        for (let i in liList) {
            liList[i].classList.add("red");
        }
    });
}

02. 계층 선택자

div p p p p p p p p p p p p
리셋 클릭하면 $("div p")를 선택합니다. 클릭하면 $("div > p")를 선택합니다. 클릭하면 $("div + p")를 선택합니다. 클릭하면 $("div ~ p")를 선택합니다. 
{
    $(".reset2").click(function () {
        $(".s2 span").removeClass("blue");
    });
    $(".btn2-1").click(function () {
        $(".s2 span").addClass("blue");
    });
    $(".btn2-2").click(function () {
        $(".s2 > span").addClass("blue");
    });
    $(".btn2-3").click(function () {
        $(".s2 .select + span").addClass("blue");
    });
    $(".btn2-4").click(function () {
        $(".s2 .select ~ span").addClass("blue");
    });    
}
